1   package logiikka;
2   /**
3    * Enum-luokka sisältää käytettävissä olevat toimialat, sekä näihin liittyvät 
4    * korrelaatiot markkinan kanssa.
5    * 
6    * HUOM: 24.5. laskettu toimialan beta-lukujen keskiarvo, joka kuvastaa tässä
7    * toimialariskiä.
8    * 
9    * @author gexgex
10   */
11  public enum Toimiala {
12      PERUSTEOLLISUUS(1.029),
13      TEOLLISUUSTUOTTEET(0.63465116),
14      KULUTUSTAVARAT(0.53285714),
15      TERVEYDENHUOLTO(0.70285714),
16      KULUTUSPALVELUT(0.475),
17      TIETOLIIKENNEPALVELUT(0.95),
18      RAHOITUS(0.48823529),
19      TEKNOLOGIA(0.62473684),
20      ENERGIA(0.975);
21      
22      private Double korrelaatioMarkkinanKanssa;
23      
24      private Toimiala(Double korrelaatio) {
25          this.korrelaatioMarkkinanKanssa = korrelaatio;
26      }
27      
28      public Double getKorrelaatio() {
29          return this.korrelaatioMarkkinanKanssa;
30      }
31      
32      /**
33       * Metodi palauttaa toimialan parametrina olevan nimen perusteella.
34       *
35       * @param merkkijono Käyttäjän syöttämä toimialan nimi ISOIN kirjaimin
36       * 
37       * @return toimiala jos sellainen merkkijonon perusteella löytyy
38       */
39      public static Toimiala merkkijonoToimialaksi(String merkkijono) {
40          for (Toimiala t : Toimiala.values()) {
41              if (merkkijono.equals(t.name())) {
42                  return t;
43              }
44          }
45          return null;
46      }
47  
48      /**
49       * Metodi palauttaa toimialan nimen isolla alkukirjaimella kirjoitettuna 
50       * String-tyyppisenä oliona.
51       */
52      @Override
53      public String toString() {
54          String toimiala = name().toLowerCase();
55          String palautettava = toimiala.substring(0, 1).toUpperCase() + toimiala.substring(1);
56          return palautettava;
57      }
58  }
