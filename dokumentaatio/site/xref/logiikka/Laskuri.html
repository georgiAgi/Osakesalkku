30      public double salkunBeta() {
31          double beta = 0;
32          int osakkeidenPainotettuMaara = 0;
33          for (Osake o : salkku.getOsakkeet()) {
34              beta += o.getMaara() * o.getHinta() * o.getRiski().getBeta();
35              osakkeidenPainotettuMaara += o.getMaara() * o.getHinta();
36          }
37          return beta / osakkeidenPainotettuMaara; //osakkeiden beta-lukujen painotettu keskiarvo;
38      }
39      
40      public int salkunOsakkeidenPainotettuMaara() {
41          int maara = 0;
42          for (Osake o : salkku.getOsakkeet()) {
43              maara += o.getMaara() * o.getHinta();
44          }
45          return maara;
46      }

48      public double salkunVolatiliteetti() {
49          double volatiliteetti = 0;
50          double osakkeidenMaara = salkunOsakkeidenPainotettuMaara();
51          for (Osake i : salkku.getOsakkeet()) {
52              for (Osake j : salkku.getOsakkeet()) {
53                  volatiliteetti += (i.getMaara() * i.getHinta() / osakkeidenMaara) * (j.getMaara() * j.getHinta() / osakkeidenMaara) * korrelaatio(i, j) * i.getRiski().getVolatiliteetti() * j.getRiski().getVolatiliteetti();
54              }
55          }
56          volatiliteetti = Math.sqrt(volatiliteetti);
57          return volatiliteetti;
58      }

60      public double korrelaatio(Osake a, Osake b) {
61          if (a.equals(b)) {
62              return 1.0;
63          }
64          return Math.min((a.getToimiala().getKorrelaatio() + b.getToimiala().getKorrelaatio()) / 2, 1.0); //korrelaatio korkeintaan 1
