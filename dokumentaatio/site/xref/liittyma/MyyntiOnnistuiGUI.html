103     public void ilmoitus() {
104         osakeLabel.setText("Myit osaketta " + vanhempi.getOsake().getNimi() + " onnistuneesti!");
105 
106         double tulos = vanhempi.getMyytyOsake().getHinta() - vanhempi.getOsake().getAlkuArvo();
107         double arvo = Math.abs(tulos * vanhempi.getMyytyOsake().getMaara());
108 
109         if (tulos < 0) {
110             tulosLabel.setText("Jäit tappiolle " + df2.format(arvo) + " euroa");
111             reaktioButton.setText("Jaaha");
112         } else if (tulos > 0) {
113             tulosLabel.setText("Jäit voitolle " + df2.format(arvo) + " euroa");
114             reaktioButton.setText("Jippii!");
115         } else {
116             tulosLabel.setText("");
117             reaktioButton.setText("Selvä");
118         }
119     }
